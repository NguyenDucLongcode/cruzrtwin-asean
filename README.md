# Setup ngay từ đầu (khung gọn nhẹ cho toàn bộ dự án)

# 1. Tạo project folder

mkdir cruzrtwin-asean
cd cruzrtwin-asean

# 2. Tạo virtual environment (QUAN TRỌNG)

python -m venv venv

# 3. Kích hoạt venv

# Windows:

venv\Scripts\activate

# Mac/Linux:

source venv/bin/activate

# 4. Tạo file requirements.txt

pip install -r requirements.txt
